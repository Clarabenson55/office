const overlay = document.getElementById('overlay');
const popupLogo = document.getElementById('popupLogo');
const popupTitle = document.getElementById('popupTitle');

// Open popup when a provider is clicked
document.querySelectorAll('.button, .button1, .button2, .button3, .button4').forEach(btn => {
  btn.addEventListener('click', () => {
    const provider = btn.dataset.name;
    popupTitle.textContent = `Sign in to ${provider}`;
    popupLogo.src = btn.querySelector('img').src;

    // Color theme change
    let color;
    switch (provider) {
      case 'Yahoo': color = '#9d1fcc'; break;
      case 'Office': color = '#eb3b00'; break;
      case 'AOL': color = '#000'; break;
      case 'Outlook': color = '#0073c6'; break;
      default: color = '#555';
    }
    document.querySelector('.popup button').style.backgroundColor = color;
    document.querySelector('.popup button').style.boxShadow = `0 0 8px ${color}`;

    overlay.style.display = 'flex';
  });
});

// Close popup (X button)

  overlay.style.display = 'none';
